def bits_to_bytes(bits):
    byte = 0
    byte_count = 0
    bytes_ = []
    for b in bits:
        byte = (byte << 1) + b
        byte_count += 1
        if byte_count == 9:
            byte -= 0x100
            bytes_.append(byte)
            byte = 0
            byte_count = 0
    return bytes_
